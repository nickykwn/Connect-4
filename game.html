var board; var currentPlayer; init(){ board = $('.spot'); currentPlayer = 'red'; } $('.spot').on('click',function(e,element){ var element = e.target; var col = element.attribute('data-column'); var validSelection = false; for (var x = 41 ; x > 0; x--; ) var currentSpot = board[x]; var currentCol = currentSpot.attr('data-column'); if (currentSpot.hasClass('empty') && currentCol == col){ currentSpot.removeClass('empty'); currentSpot.addClass(currentPlayer); validSelection = true; } if (!validSelection){ return; } if (currentPlayer == 'red'){ currentPlayer = 'blue'; }else{ currentPlayer = 'red'; } })
